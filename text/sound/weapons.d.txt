weapons {
   subweapon {
      == lsnd/WeaponsD/Subweapons
      charge Charge # Rozenkreuzstilette
      armgun ArmGun # Phantasy Star: Nova
      needle Needle # Psychic
      $alias dagger  = .weapons.rend.end
      $alias axe     = .weapons.rend.slash
      $alias grenade = .weapons.rifle.grenade
   }

   kampilan {
      == lsnd/WeaponsD/Kampilan
      draw     Draw     # Metal Gear Solid 2
      bonk     Bonk     # ^
      hit      Hit      # ^
      hitflesh HitFlesh # ^
      ready1   Ready    # ^
      ready2   Ready2   # ^
      swing1   Swing    # ^
      swing2   Swing2   # ^
      $random ready {ready1, ready2,}
      $random swing {swing1, swing2,}
   }

   \700e {
      == lsnd/WeaponsD/700Express
      fire1 Fire1      # E.Y.E: Divine Cybermancy
      fire2 Fire2      # ^
      fire3 Fire3      # ^
      out   ReloadOut  # ^
      in    ReloadIn   # ^
      move  Move       # Halo 3
      draw  Draw       # ^
      bin   BulletIn   # Temporal Assassin
      cout  CasingsOut # ^

      $random fire {
         fire1,
         fire2,
         fire3,
      }
   }

   \4bore {
      == lsnd/WeaponsD/4Bore
      fire Fire # a1337spy
      draw Draw # Sound Ideas Series 6000
   }

   shrapnel {
      == lsnd/WeaponsD/ShrapnelGun
      fire   Fire   # EYE: Divine Cybermancy
      draw   Draw   # ^
      bounce Bounce # Devil May Cry 2
      hit    Hit    # ^
   }

   motra {
      == lsnd/WeaponsD/Motra
      fire1 Fire1     # E.Y.E: Divine Cybermancy
      fire2 Fire2     # ^
      fire3 Fire3     # ^
      out   ReloadOut # ^
      in    ReloadIn  # ^
      draw  Draw      # ^
      $random fire {fire1, fire2, fire3,}
   }

   minigun {
      == lsnd/WeaponsD/Minigun
      fire   Fire        # Serious Sam: The First Encounter HD
      rotate Rotate      # ^
      up     RotateUp    # ^
      click  Click       # ^
      down   RotateDown  # E.Y.E: Divine Cybermancy
      place  RotatePlace # Quake 2
      empty  Empty       # Halo 3
      pickup Pickup      # Doom 3
      draw   Draw        # ^
   }

   dualrocket {
      == lsnd/WeaponsD/DualRocket
      firel1 FireLeft   # ???
      firel2 FireLeft2  # ???
      firel3 FireLeft3  # ???
      firer1 FireRight  # ???
      firer2 FireRight2 # ???
      firer3 FireRight3 # ???
      draw   Draw       # Doom 3
      $random firel {firel1, firel2, firel3,}
      $random firer {firer1, firer2, firer3,}
   }

   fortune {
      == lsnd/WeaponsD/FortuneGun
      fire   Fire      # Metal Gear Solid 2
      hit    Hit       # ^
      spark  Spark     # ^
      charge Charge    # Self-made
      draw   Draw      # Halo 3, Counter-Strike: Source
      $limit hit 0
      $limit spark 0
   }

   rems {
      == lsnd/WeaponsD/Rems
      fire Fire # Self-made
   }
}
